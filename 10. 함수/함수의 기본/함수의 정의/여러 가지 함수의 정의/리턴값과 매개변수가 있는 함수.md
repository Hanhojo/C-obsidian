```c
int get_factorial(int num) // int형의 값을 리턴한다.
{
	int i;
	int result = 1;

	for (i = 1; i <= num; i++) // num!을 구한다.
		result *= i;
	return result; // 구한 num!값을 리턴한다.
}
```

```c
	return PI * radius * radius;
}
```
